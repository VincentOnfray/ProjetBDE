CREATE TABLE `projetbdelocal`.`IDutilisateurs` ( `ID` INT NOT NULL AUTO_INCREMENT , PRIMARY KEY (`ID`)) ;

CREATE TABLE `projetbdelocal`.`Proposition` ( `ID` INT(255) NOT NULL AUTO_INCREMENT ,
 						`Description` VARCHAR(255) NOT NULL ,
 						`Image` VARCHAR(255) NOT NULL ,
 						`Likes` INT(255) NOT NULL DEFAULT '0' ,
 						`IDutilisateurs` INT(255) REFERENCES idutilisateurs(ID),
 						PRIMARY KEY (`ID`));

CREATE TABLE `projetbdelocal`.`Evenement` ( `ID` INT(255) NOT NULL AUTO_INCREMENT ,PRIMARY KEY (`ID`),
                                           `Description` VARCHAR(255) NOT NULL ,
                                           `Image` VARCHAR(255) NOT NULL ,
                                           `Date` DATE NOT NULL,
                                           `Recurrence` INT(255) NOT NULL,
                                           `Prix` INT(255) NOT NULL,
                                           `IDutilisateurs` INT(255) REFERENCES idutilisateurs(ID));


CREATE TABLE `projetbdelocal`.`Image` ( `ID` INT(255) NOT NULL AUTO_INCREMENT ,PRIMARY KEY (`ID`),
                                           `Image` VARCHAR(255) NOT NULL ,
                                           `Likes` INT(255) NOT NULL,
                                       	   `IDevenement` INT(255)  REFERENCES evenement(ID),
                                           `IDutilisateurs` INT(255)  REFERENCES idutilisateurs(ID));

CREATE TABLE `projetbdelocal`.`Inscription` (
                                       	   `IDevenement` INT(255) REFERENCES evenement(ID),
     									   `IDUtilisateurs` INT(255) REFERENCES idutilisateurs(idutilisateurs)
);



CREATE TABLE `projetbdelocal`.`Commentaire` (
    										`ID` INT PRIMARY KEY,
    										`Contenu` VARCHAR(255),
                                       	   `IDimage` INT(255) REFERENCES image(ID),
     									   `IDUtilisateurs` INT(255) REFERENCES idutilisateurs(idutilisateurs)
    										
);


CREATE TABLE `projetbdelocal`.`Articles` ( 
    `ID` INT(255) NOT NULL AUTO_INCREMENT  PRIMARY KEY,
    `Nom` VARCHAR(255) NOT NULL,
    `Description` VARCHAR(255) NOT NULL,
    `Prix` INT(255) not null,
    `Catégorie` VARCHAR(255),
    `image` VARCHAR(255)
    
                                         );
